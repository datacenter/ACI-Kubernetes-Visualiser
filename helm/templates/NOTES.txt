export SERVICE_IP=$(kubectl  get svc test-vkaci  --output jsonpath='{.status.loadBalancer.ingress[0].ip}')
export SERVICE_PORT=$(kubectl get svc --namespace default test-vkaci  --output jsonpath='{.spec.ports[0].port}')
